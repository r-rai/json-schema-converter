📊 Performance Benchmark

Run this in browser console after opening the tool:

// Performance test function
async function testPerformance() {
  const sizes = [1000, 10000, 50000];
  const labels = ['100KB', '1MB', '5MB'];
  
  for (let i = 0; i < sizes.length; i++) {
    const obj = {};
    for (let j = 0; j < sizes[i]; j++) {
      obj[`key${j}`] = `value${j}_${Math.random()}`;
    }
    const json = JSON.stringify(obj);
    
    console.log(`\n--- Testing ${labels[i]} (${(json.length/1024).toFixed(1)}KB) ---`);
    
    // Test Parse
    const start = performance.now();
    const parsed = JSON.parse(json);
    const parseTime = performance.now() - start;
    
    // Test Stringify
    const start2 = performance.now();
    const stringified = JSON.stringify(parsed);
    const stringifyTime = performance.now() - start2;
    
    console.log(`Parse time: ${parseTime.toFixed(2)}ms`);
    console.log(`Stringify time: ${stringifyTime.toFixed(2)}ms`);
    console.log(`Total: ${(parseTime + stringifyTime).toFixed(2)}ms`);
  }
}

testPerformance();

Target Benchmarks:

  • 100KB: < 100ms
  • 1MB: < 200ms
  • 5MB: < 300ms
